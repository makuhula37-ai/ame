            localStorage.setItem("say_expenses", JSON.stringify(dailyExpenses));
            localStorage.setItem("say_forum", JSON.stringify(forumMessages));

        // Text-to-speech flag
        let ttsEnabled = false;
        let contrastEnabled = false;

        // Helper speak
        function speakText(text) {
            if (!ttsEnabled) return;
            const utterance = new SpeechSynthesisUtterance(text);
            utterance.lang = 'en-US';
            utterance.rate = 0.9;
            window.speechSynthesis.cancel();
            window.speechSynthesis.speak(utterance);
        }

        // -------- RENDER MAIN PAGE (HOME default) ----------

            else if (currentSection === "community") renderCommunity(container);
            else if (currentSection === "about") renderAbout(container);
            else if (currentSection === "contact") renderContact(container);
            // attach extra handlers after render
            attachDynamicHandlers();
        }

        function renderExpenseList() {
            const listDiv = document.getElementById("expenseList");
            if(!listDiv) return;
            if(dailyExpenses.length===0) listDiv.innerHTML = "<p>No expenses yet.</p>";
            else {
                listDiv.innerHTML = dailyExpenses.map(exp => `<div class="expense-item"><span>${exp.desc}</span><span>$${exp.amount} (${exp.date}) <button class="del-expense" data-id="${exp.id}"><i class="fas fa-trash-alt"></i></button></span></div>`).join("");
                document.querySelectorAll(".del-expense").forEach(btn => {
                    btn.addEventListener("click", (e) => {
                        const id = btn.getAttribute("data-id");
                        dailyExpenses = dailyExpenses.filter(e => e.id !== id);
                        persistAll();
                        renderExpenseList();
                        updateTotalExpense();
                    });
                });
            }
            updateTotalExpense();
        }
        function updateTotalExpense() {
            const total = dailyExpenses.reduce((sum, e) => sum + e.amount, 0);
            const totalSpan = document.getElementById("totalExpense");
            if(totalSpan) totalSpan.innerText = `💰 Total Expenditure: $${total}`;
        }
        function addExpense() {
            const desc = document.getElementById("expenseDesc")?.value.trim();
            const amount = parseFloat(document.getElementById("expenseAmount")?.value);
            const date = document.getElementById("expenseDate")?.value;
            if(!desc || isNaN(amount)) return alert("Enter description and amount");
            const newExp = { id: Date.now().toString(), desc, amount, date };
            dailyExpenses.push(newExp);
            persistAll();
            renderExpenseList();
            document.getElementById("expenseDesc").value = "";
            document.getElementById("expenseAmount").value = "";
        }

        function drawSimpleChart() {
            const canvas = document.getElementById("progressChart");
            if(!canvas) return;
            const ctx = canvas.getContext("2d");
            canvas.width = 400; canvas.height = 200;
            ctx.clearRect(0,0,canvas.width,canvas.height);
            ctx.fillStyle = "#1e6f5c";
            ctx.fillRect(30, 150-78, 80, 78);
            ctx.fillRect(140, 150-65, 80, 65);
            ctx.fillStyle = "#f5b042";
            ctx.fillRect(250, 150-48, 80, 48);
            ctx.fillStyle = "#0f3b4f";
            ctx.font = "12px Inter";
            ctx.fillText("Chinese", 45, 170);
            ctx.fillText("AI", 160, 170);
            ctx.fillText("Both", 270, 170);
            ctx.fillStyle = "#000";
            ctx.fillText("78%", 55, 140);
            ctx.fillText("65%", 165, 145);
            ctx.fillText("48%", 280, 115);
        }

                    inputChat.value = "";
                }
            };

        function renderForumMessages() {
            const forumDiv = document.getElementById("forumMessagesList");
            if(!forumDiv) return;
            const selectedCourse = document.getElementById("forumCourseSelect")?.value || "Chinese Language";
            const filtered = forumMessages.filter(f => f.course === selectedCourse).slice(-8);
            forumDiv.innerHTML = filtered.map(m => `<div class="forum-post"><strong>${m.user}</strong>: ${m.message}<br><span class="badge">${m.timestamp}</span></div>`).join("");
            if(filtered.length===0) forumDiv.innerHTML = "<p>No messages yet. Start discussion!</p>";
            document.getElementById("forumCourseSelect")?.addEventListener("change", () => renderForumMessages());
        }

        function attachDynamicHandlers() {
            document.querySelectorAll(".speak-btn").forEach(btn => {
                btn.addEventListener("click", () => { const txt = btn.getAttribute("data-text"); if(txt) speakText(txt); });
            });
            document.getElementById("toggleContrastBtn")?.addEventListener("click", () => {
                contrastEnabled = !contrastEnabled;
                if(contrastEnabled) document.body.classList.add("high-contrast");
                else document.body.classList.remove("high-contrast");
            });
            document.getElementById("ttsToggleBtn")?.addEventListener("click", () => {
                ttsEnabled = !ttsEnabled;
                alert(`Text-to-Speech ${ttsEnabled ? "ON" : "OFF"}`);
            });
        }

            });
